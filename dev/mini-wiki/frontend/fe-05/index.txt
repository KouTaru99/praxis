2c:Tc00,import { useEffect, useState } from 'react';
import type { ArticleListItem, Paginated, Tag } from '../types';
import { listArticles, searchArticles, listTags } from '../lib/api';
import ArticleCard from '../components/ArticleCard';

type DataState =
  | { status: 'loading' }
  | { status: 'error'; message: string }
  | { status: 'ok'; result: Paginated<ArticleListItem> };

export default function ArticleListPage() {
  const [query, setQuery] = useState('');
  const [activeTag, setActiveTag] = useState<string | null>(null);
  const [tags, setTags] = useState<Tag[]>([]);
  const [data, setData] = useState<DataState>({ status: 'loading' });

  useEffect(() => { listTags().then(setTags).catch(() => {}); }, []);

  // Fetch — debounce 300ms khi đang tìm kiếm
  useEffect(() => {
    let cancelled = false;
    const q = query.trim();
    const doFetch = () => {
      const promise = q ? searchArticles(q) : listArticles(activeTag ? { tag: activeTag } : undefined);
      promise
        .then((result) => { if (!cancelled) setData({ status: 'ok', result }); })
        .catch((err: unknown) => {
          if (!cancelled)
            setData({ status: 'error', message: err instanceof Error ? err.message : 'Lỗi' });
        });
    };
    setData({ status: 'loading' });
    if (q) {
      const timer = setTimeout(doFetch, 300); // debounce
      return () => { cancelled = true; clearTimeout(timer); };
    }
    doFetch();
    return () => { cancelled = true; };
  }, [query, activeTag]);

  // Tag và search loại trừ nhau
  const handleTagClick = (slug: string) => {
    setActiveTag((prev) => (prev === slug ? null : slug));
    setQuery('');
  };
  const handleQueryChange = (q: string) => {
    setQuery(q);
    if (q.trim()) setActiveTag(null);
  };

  return (
    <div>
      <h1 className="page-title">Bài viết</h1>
      <div className="search-bar">
        <input className="search-input" type="search" value={query}
          onChange={(e) => handleQueryChange(e.target.value)} placeholder="Tìm kiếm bài viết…" />
      </div>
      {tags.length > 0 && (
        <div className="tag-filter">
          {tags.map((tag) => (
            <button key={tag.id} type="button"
              className={`tag-chip tag-chip--btn${activeTag === tag.slug ? ' tag-chip--active' : ''}`}
              onClick={() => handleTagClick(tag.slug)}>
              {tag.name}
            </button>
          ))}
        </div>
      )}
      {data.status === 'loading' && <p className="state-msg">Đang tải…</p>}
      {data.status === 'error' && <p className="state-msg error">Lỗi: {data.message}</p>}
      {data.status === 'ok' &&
        (data.result.data.length === 0 ? (
          <p className="state-msg">
            {query.trim() ? `Không tìm thấy kết quả cho "${query.trim()}".`
              : activeTag ? 'Không có bài viết nào với thẻ này.' : 'Chưa có bài viết nào.'}
          </p>
        ) : (
          data.result.data.map((a) => <ArticleCard key={a.id} article={a} />)
        ))}
    </div>
  );
}
