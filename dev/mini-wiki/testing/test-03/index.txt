# Kiểm tra toàn bộ API mini-wiki end-to-end. Chạy: bash scripts/smoke-test.sh
set -euo pipefail
BASE_URL="${BASE_URL:-http://localhost:3000}"
PASS=0; FAIL=0

assert_status() { # assert_status <mong-đợi> <thực-tế> <mô-tả>
  if [ "$2" -eq "$1" ]; then
    echo "  ✓ $3 (HTTP $2)"; PASS=$((PASS + 1))
  else
    echo "  ✗ $3 — mong đợi $1, nhận $2"; FAIL=$((FAIL + 1))
  fi
}

# API có sống không?
if ! curl -sf "$BASE_URL/health" > /dev/null; then
  echo "Lỗi: API không phản hồi. Hãy chạy: docker compose up -d db && cd api && npm run dev"
  exit 1
fi

TS=$(date +%s)                       # timestamp -> dữ liệu test không trùng
TITLE="Smoke Test Article $TS"

# Ví dụ 2 trong 12 kịch bản:
code=$(curl -s -o /dev/null -w '%{http_code}' "$BASE_URL/api/articles")
assert_status 200 "$code" "GET danh sách bài viết"

code=$(curl -s -o /dev/null -w '%{http_code}' -X POST "$BASE_URL/api/articles" \
  -H 'Content-Type: application/json' -d "{\"title\":\"$TITLE\",\"content\":\"# Hi\"}")
assert_status 201 "$code" "POST tạo bài viết mới"

# ... 10 kịch bản còn lại: get-by-slug, update, search, tag CRUD,
#     xóa tag đang dùng (409), delete bài, 404 sau khi xóa ...

echo ""; echo "Kết quả: $PASS pass, $FAIL fail"
[ "$FAIL" -eq 0 ] || exit 1
